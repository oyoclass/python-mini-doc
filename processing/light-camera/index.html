Example

from processing import *

def setup():
    size(560, 400, P3D)
    #printCamera()
    #printProjection()

def draw():
    noFill()
    beginCamera()
    camera()
    rotateX(-PI/6)
    endCamera()

    translate(50, 50, 0)
    rotateY(PI/3)
    box(45)


run()
