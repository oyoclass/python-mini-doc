Example

from processing import *

myFont = None

def setup():
    global myFont
    size(400, 400)
    background(0)
    myFont = createFont("Georgia", 32);

def draw():
    fill(255, 255, 255)
    textFont(myFont)
    textAlign(LEFT)
    text("Hello World", 100, 200)

    fill(255, 0, 0)
    textSize(30)
    text("Start Game", 100, 300)


run()
